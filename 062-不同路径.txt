class Solution {
public:
    int uniquePaths(int m, int n) {
        if(m==1||n==1)
            return 1;
         vector<vector<int> > res(n, vector<int>(m));
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            if(i==0)
                res[i][j]=1;
            else if(j==0)
                res[i][j]=res[i-1][j];
            else
                res[i][j]=res[i][j-1]+res[i-1][j];
        }
    }
    return res[n-1][m-1];
    }
   
};//Ë®Ìâ~
