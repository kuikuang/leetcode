class Solution {
public:
    int maxSubArray(vector<int>& nums) {
         int sum=nums[0];
         int n=nums[0];
    
        for(int i=1;i<nums.size();i++)
        {
          if(n>0)
             n=n+nums[i];
            else
                n=nums[i];
            if(sum<n)
                sum=n;
        }
         
          return sum;
   
    }
};
